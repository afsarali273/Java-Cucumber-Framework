Test Runners

Overview

Sands Framework supports two test execution approaches: TestNG Runner (for local development) and CLI Runner (for CI/CD pipelines).

1Ô∏è‚É£ TestNG Runner

TestNG runner is ideal for local development with IDE support and easy debugging.

Implementation

package com.automation.runners;

import io.cucumber.testng.AbstractTestNGCucumberTests;
import io.cucumber.testng.CucumberOptions;
import org.testng.annotations.DataProvider;

@CucumberOptions(
    features = "src/test/resources/features",
    glue = {"com.automation.core.commonSteps", "com.automation.stepdefinitions"},
    tags = "@UI",
    plugin = {
        "pretty",
        "html:test-output/cucumber-reports/cucumber.html",
        "json:test-output/cucumber-reports/cucumber.json",
        "io.qameta.allure.cucumber7jvm.AllureCucumber7Jvm"
    },
    monochrome = true,
    dryRun = false
)
public class TestNGCucumberRunner extends AbstractTestNGCucumberTests {
    
    @Override
    @DataProvider(parallel = true)
    public Object[][] scenarios() {
        return super.scenarios();
    }
}

Tag Expressions

// Single tag
tags = "@UI"

// Multiple tags (AND)
tags = "@UI and @Smoke"

// Multiple tags (OR)
tags = "@UI or @API"

// Exclude tags
tags = "@UI and not @Skip"

// Complex expressions
tags = "(@UI or @API) and @Smoke and not @WIP"

Running TestNG Runner

From IDE (IntelliJ/Eclipse)

  • Right-click on TestNGCucumberRunner class
  • Select "Run TestNGCucumberRunner"
  • View results in IDE test panel

From Maven

# Run all tests
mvn clean test

# Run specific tags
mvn test -Dcucumber.filter.tags="@UI"

# Run with specific browser
mvn test -Dbrowser=firefox -Dcucumber.filter.tags="@Smoke"

# Run with thread count
mvn test -Dthread.count=3

Using testng.xml

<?xml version="1.0" encoding="UTF-8"?>

    </test>

# Run with testng.xml
mvn test -DsuiteXmlFile=testng.xml

2Ô∏è‚É£ CLI Runner (Cucumber Main)

CLI runner is lightweight and perfect for CI/CD pipelines with faster startup time.

Implementation

package com.automation.runners;

import io.cucumber.core.cli.Main;
import java.io.InputStream;
import java.util.Objects;
import java.util.Properties;

public class CucumberCLIRunner {
    
    public static void main(String[] args) {
        String threadCount = getThreadCount();
        String myTag = "@UI";
        String tags = Objects.isNull(getTags()) ? myTag : getTags();
        
        String[] baseArgs = {
            "--glue", "com.automation.core.commonSteps",
            "--glue", "com.automation.stepdefinitions", 
            "--plugin", "pretty",
            "--plugin", "html:test-output/cucumber-reports/cucumber.html",
            "--plugin", "json:test-output/cucumber-reports/cucumber.json",
            "--plugin", "io.qameta.allure.cucumber7jvm.AllureCucumber7Jvm",
            "--threads", threadCount
        };
        
        String[] defaultArgs = tags != null ? 
            addTags(baseArgs, tags) : 
            addFeaturePath(baseArgs);
        
        String[] finalArgs = args.length > 0 ? args : defaultArgs;
        Main.main(finalArgs);
    }
    
    private static String getThreadCount() {
        String systemProperty = System.getProperty("thread_count");
        if (systemProperty != null) {
            return systemProperty;
        }
        
        try (InputStream input = CucumberCLIRunner.class.getClassLoader()
                .getResourceAsStream("config.properties")) {
            Properties prop = new Properties();
            prop.load(input);
            return prop.getProperty("thread.count", "1");
        } catch (Exception e) {
            return "1";
        }
    }
    
    private static String getTags() {
        return System.getProperty("cucumber.filter.tags");
    }
    
    private static String[] addTags(String[] baseArgs, String tags) {
        String[] result = new String[baseArgs.length + 3];
        System.arraycopy(baseArgs, 0, result, 0, baseArgs.length);
        result[baseArgs.length] = "--tags";
        result[baseArgs.length + 1] = tags;
        result[baseArgs.length + 2] = "src/test/resources/features";
        return result;
    }
    
    private static String[] addFeaturePath(String[] baseArgs) {
        String[] result = new String[baseArgs.length + 1];
        System.arraycopy(baseArgs, 0, result, 0, baseArgs.length);
        result[baseArgs.length] = "src/test/resources/features";
        return result;
    }
}

Running CLI Runner

From Command Line

# Run with default tags (@UI)
java -cp target/classes:target/test-classes com.automation.runners.CucumberCLIRunner

# Run with specific tags
java -Dcucumber.filter.tags="@Smoke" -cp target/classes:target/test-classes \
  com.automation.runners.CucumberCLIRunner

# Run with thread count
java -Dthread_count=5 -Dcucumber.filter.tags="@API" \
  -cp target/classes:target/test-classes com.automation.runners.CucumberCLIRunner

From Maven with Custom Profile

# Add to pom.xml
<profiles>
    <profile>
        <id>cli-runner</id>
        <build>
            <plugins>
                <plugin>

                    <executions>
                        <execution>
                            <phase>test</phase>
                            <goals>
                                <goal>java</goal>
                            </goals>
                        </execution>
                    </executions>
                    <configuration>
                        <mainClass>com.automation.runners.CucumberCLIRunner</mainClass>
                        <classpathScope>test</classpathScope>
                    </configuration>
                </plugin>
            </plugins>
        </build>
    </profile>
</profiles>

Dynamic Configuration

public class CucumberCLIRunner {
    
    public static void main(String[] args) {
        List<String> argsList = new ArrayList<>();
        
        // Base configuration
        argsList.addAll(Arrays.asList(
            "--glue", "com.automation.core.commonSteps",
            "--glue", "com.automation.stepdefinitions"
        ));
        
        // Add plugins
        addPlugins(argsList);
        
        // Add tags from system property or default
        String tags = System.getProperty("cucumber.filter.tags", "@UI");
        argsList.add("--tags");
        argsList.add(tags);
        
        // Add thread count
        String threads = System.getProperty("thread_count", "1");
        argsList.add("--threads");
        argsList.add(threads);
        
        // Add feature path
        String featurePath = System.getProperty("feature.path", 
            "src/test/resources/features");
        argsList.add(featurePath);
        
        // Execute
        Main.main(argsList.toArray(new String[0]));
    }
    
    private static void addPlugins(List<String> argsList) {
        argsList.add("--plugin");
        argsList.add("pretty");
        argsList.add("--plugin");
        argsList.add("html:test-output/cucumber-reports/cucumber.html");
        argsList.add("--plugin");
        argsList.add("json:test-output/cucumber-reports/cucumber.json");
        argsList.add("--plugin");
        argsList.add("io.qameta.allure.cucumber7jvm.AllureCucumber7Jvm");
    }
}

CI/CD Integration

Jenkins Pipeline (CLI Runner)

pipeline {
    agent any
    
    parameters {
        choice(name: 'TAGS', choices: ['@UI', '@API', '@Smoke', '@Regression'], 
               description: 'Select test tags')
        choice(name: 'BROWSER', choices: ['chrome', 'firefox', 'edge'], 
               description: 'Select browser')
        string(name: 'THREAD_COUNT', defaultValue: '3', 
               description: 'Number of parallel threads')
    }
    
    stages {
        stage('Checkout') {
            steps {
                git branch: 'main', url: 'https://github.com/your-repo/SandsAutoFramework.git'
            }
        }
        
        stage('Build') {
            steps {
                sh 'mvn clean compile'
            }
        }
        
        stage('Run Tests - CLI Runner') {
            steps {
                sh """
                    mvn clean test -Pcli-runner \
                    -Dcucumber.filter.tags="${params.TAGS}" \
                    -Dbrowser=${params.BROWSER} \
                    -Dthread_count=${params.THREAD_COUNT}
                """
            }
        }
        
        stage('Generate Reports') {
            steps {
                allure includeProperties: false, 
                       jdk: '', 
                       results: [[path: 'target/allure-results']]
                
                publishHTML([
                    allowMissing: false,
                    alwaysLinkToLastBuild: true,
                    keepAll: true,
                    reportDir: 'test-output/cucumber-reports',
                    reportFiles: 'cucumber.html',
                    reportName: 'Cucumber Report'
                ])
            }
        }
    }
    
    post {
        always {
            junit 'test-output/cucumber-reports/*.xml'
            archiveArtifacts artifacts: 'test-output/**/*', allowEmptyArchive: true
        }
    }
}

on:

  test:
    runs-on: ubuntu-latest

    - name: Checkout code
      uses: actions/checkout@v3
    
    - name: Set up JDK 11
      uses: actions/setup-java@v3
      with:
        java-version: '11'
        distribution: 'temurin'
    
    - name: Cache Maven packages
      uses: actions/cache@v3
      with:
        path: ~/.m2
        key: ${{ runner.os }}-m2-${{ hashFiles('**/pom.xml') }}

      with:
        allure_results: target/allure-results
        allure_history: allure-history

      if: always()
      uses: actions/upload-artifact@v3
      with:
        name: test-reports
        path: |

Local development ‚úÖ TestNG Runner IDE support, easy debugging
Debugging test failures ‚úÖ TestNG Runner Breakpoint debugging in IDE
Running from IntelliJ/Eclipse ‚úÖ TestNG Runner Native IDE integration
CI/CD pipelines ‚úÖ CLI Runner Faster, lightweight, less overhead
Jenkins/GitHub Actions ‚úÖ CLI Runner Better for automation servers

Regression suite Either Both work well

Best Practices

General Best Practices

  • ‚úÖ Use TestNG runner for local development and debugging
  • ‚úÖ Use CLI runner for CI/CD pipelines and automation servers
  • ‚úÖ Keep both runners in sync with same configuration
  • ‚úÖ Use tags to organize and filter tests
  • ‚úÖ Configure parallel execution based on available resources
  • ‚úÖ Generate multiple report formats (HTML, JSON, Allure)
  • ‚úÖ Archive test reports in CI/CD for historical analysis
  • ‚úÖ Use meaningful tag names (@Smoke, @Regression, @Critical)

TestNG Runner Best Practices

  • Use testng.xml for complex test suites
  • Enable parallel execution with @DataProvider(parallel = true)
  • Create separate runners for different test categories
  • Use TestNG groups for additional test organization
  • Configure retry logic for flaky tests

CLI Runner Best Practices

  • Read configuration from config.properties
  • Support system properties for runtime overrides
  • Provide sensible defaults for all parameters
  • Use Maven profiles for different execution modes
  • Keep CLI runner lightweight and fast
  • Log configuration at startup for debugging

CI/CD Best Practices

  • Run smoke tests on every commit
  • Run regression tests nightly or on schedule
  • Use headless mode for faster execution
  • Cache Maven dependencies to speed up builds
  • Fail fast on critical test failures
  • Generate and publish test reports automatically
  • Send notifications on test failures
  • Track test execution trends over time

Troubleshooting

Common Issues

Issue: Tests not running with CLI Runner

// Solution: Ensure classpath includes test classes
java -cp "target/classes:target/test-classes:target/dependency/*" \
  com.automation.runners.CucumberCLIRunner

Issue: Tags not working

// Solution: Use correct tag expression syntax
--tags "@UI and @Smoke"  // ‚úÖ Correct
--tags "@UI @Smoke"      // ‚ùå Wrong

Issue: Parallel execution not working

// TestNG: Enable parallel in @DataProvider
@DataProvider(parallel = true)
public Object[][] scenarios() {
    return super.scenarios();
}

// CLI: Use --threads option
--threads 3

Issue: Reports not generated

// Solution: Ensure output directory exists
mkdir -p test-output/cucumber-reports

// Check plugin configuration
--plugin html:test-output/cucumber-reports/cucumber.html

Issue: Step definitions not found

// Solution: Verify glue packages
--glue com.automation.core.commonSteps
--glue com.automation.stepdefinitions
